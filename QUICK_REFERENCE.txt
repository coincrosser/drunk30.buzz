// Quick Reference: Commands to Get drunk30.buzz Running

// ============================================
// INITIAL SETUP (Do once)
// ============================================

// 1. Install dependencies
npm install

// 2. Set up database
npm run prisma:migrate

// 3. Start development server
npm run dev
// → Opens at http://localhost:3000


// ============================================
// DAILY DEVELOPMENT
// ============================================

// Start dev server (run this first every time)
npm run dev

// In another terminal, test generators
bash test-generators.sh

// View database GUI
npm run prisma:studio


// ============================================
// WHEN SOMETHING BREAKS
// ============================================

// Full reset
rm -rf node_modules package-lock.json .next
npm install
npm run prisma:migrate
npm run dev

// Just the database
npm run prisma:migrate

// Just the server cache
rm -rf .next
npm run dev


// ============================================
// BUILDING FOR PRODUCTION
// ============================================

// Build the project
npm run build

// Start production server
npm run start


// ============================================
// DEBUGGING TIPS
// ============================================

// Check Node version (needs >= 18)
node -v

// Test a generator endpoint
curl -X POST http://localhost:3000/api/ai/outline \
  -H "Content-Type: application/json" \
  -d '{"topic":"Test Topic"}'

// Check if .env.local exists
ls -la .env.local

// View server logs
npm run dev  // Shows all logs in terminal

// Browser console
F12 → Console tab
